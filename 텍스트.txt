function fmtCur(v, market) {
  const abs = Math.abs(v);
  if (market === 'US') return '$' + abs.toLocaleString('en-US',{minimumFractionDigits:2,maximumFractionDigits:2});
  return '₩' + abs.toLocaleString('ko-KR');
}

// Tab
function switchTab(tab, btn) {
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  ['list','stats','chart'].forEach(t => {
    document.getElementById('tab-'+t).style.display = t === tab ? '' : 'none';
  });
  currentTab = tab;
  renderAll();
}

function setFilter(f, btn) {
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  currentFilter = f;
  renderList();
}

function renderAll() {
  renderList();
  renderStats();
  renderChart();
}

// LIST
function renderList() {
  const filtered = currentFilter === 'ALL' ? trades : trades.filter(t => t.market === currentFilter);
  const el = document.getElementById('trade-list');
  if (!filtered.length) {
    el.innerHTML = `<div class="empty"><div class="empty-icon">📋</div><div>매매 기록이 없어요</div><div style="font-size:13px;margin-top:6px">+ 버튼을 눌러 첫 기록을 추가해보세요</div></div>`;
    return;
  }
  el.innerHTML = filtered.map(t => {
    const total = fmtCur(t.price * t.quantity, t.market);
    const memoHtml = t.memo ? `<div class="trade-memo">${t.memo.length > 60 ? t.memo.slice(0,60)+'...' : t.memo}</div>` : '';
    return `
    <div class="trade-card ${t.type}" onclick="openDetail(${t.id})">
      <div class="trade-row">
        <div>
          <div class="trade-name">${t.name}<span class="trade-ticker">${t.ticker}</span></div>
          <div class="trade-badges">
            <span class="badge ${t.type==='buy'?'green':'red'}">${t.type==='buy'?'📈 매수':'📉 매도'}</span>
            <span class="badge">${t.market==='KR'?'🇰🇷':'🇺🇸'}</span>
            <span class="trade-date">${t.date}</span>
          </div>
        </div>
        <div>
          <div class="trade-amount">${total}</div>
          <div class="trade-detail">${t.price.toLocaleString()} × ${t.quantity}주</div>
        </div>
      </div>
      ${memoHtml}
      <div class="trade-emotion">${t.emotion}</div>
    </div>`;
  }).join('');
}

// STATS
function renderStats() {
  const kr = trades.filter(t => t.market === 'KR');
  const us = trades.filter(t => t.market === 'US');
  const pnl = list => list.reduce((s,t) => {
    if (t.type === 'sell') return s + t.price*t.quantity - t.fee;
    return s - t.price*t.quantity - t.fee;
  }, 0);
  const krPnl = pnl(kr), usPnl = pnl(us);
  const emotionCount = {};
  trades.forEach(t => { emotionCount[t.emotion] = (emotionCount[t.emotion]||0)+1; });
  const topEmotion = Object.entries(emotionCount).sort((a,b)=>b[1]-a[1])[0]?.[0] || '-';

  const emotionRows = EMOTIONS.map(em => {
    const cnt = emotionCount[em] || 0;
    if (!cnt) return '';
    const pct = trades.length ? (cnt/trades.length*100).toFixed(0) : 0;
    return `<div class="emotion-row">
      <div class="emotion-row-top"><span>${em}</span><span style="color:#666;font-size:12px">${cnt}회 (${pct}%)</span></div>
      <div class="emotion-bar-bg"><div class="emotion-bar-fill" style="width:${pct}%"></div></div>
    </div>`;
  }).join('');

  document.getElementById('stats-content').innerHTML = `
    <div class="stat-grid">
      <div class="stat-card"><div class="stat-label">총 매매 횟수</div><div class="stat-value">${trades.length}회</div></div>
      <div class="stat-card"><div class="stat-label">매수 / 매도</div><div class="stat-value">${trades.filter(t=>t.type==='buy').length} / ${trades.filter(t=>t.type==='sell').length}</div></div>
    </div>
    <div class="pnl-card">
      <div class="pnl-label">국내 주식 손익 (KRW)</div>
      <div class="pnl-value ${krPnl>=0?'pos':'neg'}">${krPnl>=0?'+':'-'}₩${Math.abs(krPnl).toLocaleString()}</div>
    </div>
    <div class="pnl-card">
      <div class="pnl-label">해외 주식 손익 (USD)</div>
      <div class="pnl-value ${usPnl>=0?'pos':'neg'}">${usPnl>=0?'+':'-'}$${Math.abs(usPnl).toFixed(2)}</div>
    </div>
    <div class="stat-card" style="margin-bottom:12px">
      <div class="stat-label">가장 많은 심리 상태</div>
      <div style="font-size:24px;margin-top:6px">${topEmotion}</div>
    </div>
    <div class="emotion-card">
      <div class="pnl-label">심리 상태 분포</div>
      ${emotionRows || '<div style="color:#444;font-size:13px">기록이 없어요</div>'}
    </div>
  `;
}

// CHART
function renderChart() {
  const sorted = [...trades].filter(t=>t.market==='KR').sort((a,b)=>a.date.localeCompare(b.date));
  const byDate = {};
  sorted.forEach(t => {
    byDate[t.date] = (byDate[t.date]||0) + (t.type==='sell' ? t.price*t.quantity : -t.price*t.quantity);
  });
  const entries = Object.entries(byDate);
  let running = 0;
  const chartData = entries.map(([date,val]) => { running+=val; return {date, val:running}; });

  let svgContent = '<div style="text-align:center;padding:30px 0;color:#444;font-size:13px">데이터가 부족합니다</div>';
  if (chartData.length >= 2) {
    const maxAbs = Math.max(...chartData.map(d=>Math.abs(d.val)),1);
    const pts = chartData.map((d,i) => {
      const x = (i/(chartData.length-1))*290+5;
      const y = 60-(d.val/maxAbs)*50;
      return `${x},${y}`;
    }).join(' ');
    const dots = chartData.map((d,i) => {
      const x = (i/(chartData.length-1))*290+5;
      const y = 60-(d.val/maxAbs)*50;
      return `<circle cx="${x}" cy="${y}" r="4" fill="#c8f542"/>`;
    }).join('');
    const labels = chartData.map((d,i) => {
      if (i % Math.max(1,Math.floor(chartData.length/3)) !== 0 && i !== chartData.length-1) return '';
      const x = (i/(chartData.length-1))*290+5;
      return `<text x="${x}" y="115" text-anchor="middle" font-size="9" fill="#555">${d.date.slice(5)}</text>`;
    }).join('');
    svgContent = `<svg class="chart" viewBox="0 0 300 120">
      <line x1="0" y1="60" x2="300" y2="60" stroke="#333" stroke-width="1"/>
      <polyline points="${pts}" fill="none" stroke="#c8f542" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"/>
      ${dots}${labels}
    </svg>`;
  }

  const timeline = [...trades].sort((a,b)=>b.date.localeCompare(a.date)).map((t,i,arr) => {
    const lineHtml = i < arr.length-1 ? '<div class="timeline-line"></div>' : '';
    return `<div class="timeline-item">
      <div class="timeline-dot-col">
        <div class="timeline-dot ${t.type}"></div>
        ${lineHtml}
      </div>
      <div>
        <div class="timeline-date">${t.date}</div>
        <div class="timeline-name">${t.name} <span class="${t.type==='buy'?'timeline-type-buy':'timeline-type-sell'}">${t.type==='buy'?'매수':'매도'}</span></div>
        <div class="timeline-sub">${fmtCur(t.price*t.quantity,t.market)} · ${t.emotion}</div>
      </div>
    </div>`;
  }).join('');

  document.getElementById('chart-content').innerHTML = `
    <div class="chart-card">
      <div class="chart-label">국내 주식 누적 손익 추이 (KRW)</div>
      ${svgContent}
    </div>
    <div class="timeline-card">
      <div class="pnl-label">매매 타임라인</div>
      ${timeline || '<div style="color:#444;font-size:13px">기록이 없어요</div>'}
    </div>
  `;
}

// MODAL
function openAddModal() {
  const today = new Date().toISOString().split('T')[0];
  document.getElementById('f-date').value = today;
  document.getElementById('f-ticker').value = '';
  document.getElementById('f-name').value = '';
  document.getElementById('f-price').value = '';
  document.getElementById('f-qty').value = '';
  document.getElementById('f-fee').value = '';
  document.getElementById('f-memo').value = '';
  document.getElementById('f-market').value = 'KR';
  document.getElementById('f-type').value = 'buy';
  selectedEmotion = '😄 신남';
  document.querySelectorAll('.emotion-btn').forEach(b => b.classList.toggle('active', b.textContent === selectedEmotion));
  document.getElementById('add-modal').classList.add('open');
}

function closeModal(id) { document.getElementById(id).classList.remove('open'); }

function selectEmotion(btn, em) {
  selectedEmotion = em;
  document.querySelectorAll('.emotion-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
}

function addTrade() {
  const ticker = document.getElementById('f-ticker').value.trim().toUpperCase();
  const name = document.getElementById('f-name').value.trim();
  const price = parseFloat(document.getElementById('f-price').value);
  const qty = parseInt(document.getElementById('f-qty').value);
  if (!ticker || !name || !price || !qty) { alert('종목코드, 종목명, 가격, 수량은 필수입니다.'); return; }
  const trade = {
    id: Date.now(),
    date: document.getElementById('f-date').value,
    ticker, name,
    type: document.getElementById('f-type').value,
    price, quantity: qty,
    market: document.getElementById('f-market').value,
    emotion: selectedEmotion,
    memo: document.getElementById('f-memo').value.trim(),
    fee: parseFloat(document.getElementById('f-fee').value) || 0,
  };
  trades.unshift(trade);
  saveTrades(trades);
  closeModal('add-modal');
  renderAll();
}

function openDetail(id) {
  const t = trades.find(t => t.id === id);
  if (!t) return;
  selectedTradeId = id;
  document.getElementById('detail-title').textContent = `${t.name} (${t.ticker})`;
  const memoHtml = t.memo ? `<div class="detail-section"><div class="detail-key">일기 / 메모</div><div class="memo-text">${t.memo}</div></div>` : '';
  document.getElementById('detail-body').innerHTML = `
    <div style="display:flex;gap:8px;margin-bottom:12px">
      <span class="badge ${t.type==='buy'?'green':'red'}">${t.type==='buy'?'📈 매수':'📉 매도'}</span>
      <span class="badge">${t.market==='KR'?'🇰🇷 국내':'🇺🇸 해외'}</span>
      <span class="badge">${t.date}</span>
    </div>
    <div class="detail-section">
      <div class="detail-grid">
        <div><div class="detail-key">가격</div><div class="detail-val">${fmtCur(t.price,t.market)}</div></div>
        <div><div class="detail-key">수량</div><div class="detail-val">${t.quantity}주</div></div>
        <div><div class="detail-key">총금액</div><div class="detail-val">${fmtCur(t.price*t.quantity,t.market)}</div></div>
        <div><div class="detail-key">수수료</div><div class="detail-val">${fmtCur(t.fee,t.market)}</div></div>

    <button class="delete-btn" onclick="deleteTrade(${id})">🗑 기록 삭제</button>
  `;
  document.getElementById('detail-modal').classList.add('open');
}

function deleteTrade(id) {
  if (!confirm('이 기록을 삭제할까요?')) return;
  trades = trades.filter(t => t.id !== id);
  saveTrades(trades);
  closeModal('detail-modal');
  renderAll();
}
